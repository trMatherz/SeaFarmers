# Breed Counting Solution

Tags: Prefix Sums

# Breakdown

Maintain a separate prefix array for each breed. 

This runs in $\mathcal{O}(n)$

# Implementation

<div style={{ transform: 'translateX(150px)' }}>
  <SyntaxHighlighter language="cpp" style={okaidia}>
{`#include <cmath>
#include <cstdio>
#include <vector>
#include <fstream> 
std::ifstream cin("bcount.in");
std::ofstream cout("bcount.out");
#include <algorithm>
using namespace std;

int main() {
    int n, q; cin >> n >> q; 
    vector<vector<int>> a(3, vector<int>(n, 0)); 
    for(int i = 0; i < n; i++) { int x; cin >> x; a[--x][i] = 1; }
    vector<vector<int>> pre(3, vector<int>(n + 1, 0)); 
    for(int j = 0; j < 3; j++) for(int i = 0; i < n; i++) pre[j][i + 1] = pre[j][i] + a[j][i]; 
    for(int i = 0; i < q; i++) {
        int x, y; cin >> x >> y; 
        x--, y--; 
        cout << pre[0][y + 1] - pre[0][x] << " "; 
        cout << pre[1][y + 1] - pre[1][x] << " "; 
        cout << pre[2][y + 1] - pre[2][x] << endl;
    }
}
`}
