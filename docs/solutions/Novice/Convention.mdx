# Convention Solution

Tags: Binary Search

# Breakdown

We can test a certain time in $\mathcal{O}(n)$, and the function is monotonic. 

This runs in $\mathcal{O}(\log_2(n)*n)$

{`#include <cmath>
#include <cstdio>
#include <vector>
#include <fstream> 
std::ifstream cin("convention.in");
std::ofstream cout("convention.out");
#include <algorithm>
using namespace std;

int n, k, p; 
vector<int> a; 
int check(int x) {  
    int last = a[0], ct = 1; 
    int ret = 1; 
    for(int i = 1; i < n; i++) {
        if(ct == p || last + x < a[i]) ret++, last = a[i], ct = 1; 
        else ct++; 
    }
    return ret; 
}
int main() {
    cin >> n >> k >> p; 
    int l = 0, r = (int) 1e9 + 7;
    a = vector<int>(n); 
    for(int i = 0; i < n; i++) cin >> a[i]; 
    sort(a.begin(), a.end()); 
    while(l < r) {
        int m = (l + r) / 2; 
        if(check(m) <= k) r = m; 
        else l = m + 1; 
    }
    cout << l; 
}
