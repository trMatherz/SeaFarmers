# Breakdown

This is a simulation problem, so all you have to do is translate the problem into code form. You can do this 
with If-Statements and variables, but I used a map to simplify.

#include <algorithm>
#include <map>
using namespace std;


int main() {
    int n; 
    cin >> n; 
    map<char, long long> cur; 
    for(int i = 0; i < n; i++) {
        int k; 
        cin >> k; 
        for(int j = 0; j < k; j++) {
            string w; long long x; char y; 
            cin >> w >> x >> y; 
            if(w == "Grow") cur[y] += x; 
            else if(w == "Die") cur[y] -= x; 
            else {
                char z; 
                cin >> z; 
                cur[y] -= x; 
                cur[z] += x; 
            }
        }
        cout << cur['R'] << " " << 'R' << " ";
        cout << cur['B'] << " " << 'B' << " ";
        cout << cur['P'] << " " << 'P' << endl; 
    }
}
