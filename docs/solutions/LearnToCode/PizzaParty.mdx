# Breakdown

We can simulate this process using a For Loop. 

This runs in $\mathcal{O}(n)$

# Implementation

<div style={{ transform: 'translateX(150px)' }}>
  <SyntaxHighlighter language="cpp" style={okaidia}>
{`#include <cmath>
#include <cstdio>
#include <vector>
#include <fstream>
std::ifstream cin ("paint.in");
std::ofstream cout ("paint.out");
#include <algorithm>
using namespace std;


int main() {
    int n; cin >> n; 
    for(int i = 0; i < n; i++) {
        int s, f; cin >> s >> f; 
        int num = s / f; 
        int rem = s % f; 
        cout << "All friends shall get " << num << " slice";
        if (num != 1) cout << "s"; 
        cout << " and Mark shall get " << rem / 2 << " slice";
        if (rem / 2 != 1) cout << "s"; 
        cout << " which leaves Jerry with " << (rem + 1) / 2 << " slice";
        if ((rem + 1) / 2 != 1) cout << "s"; 
        cout << "!\n"; 
    }
    return 0;
}
`}
